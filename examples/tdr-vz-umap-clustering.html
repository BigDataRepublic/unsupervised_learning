from encoders import WeightOfEvidenceEncoder


def make_pipeline(sampler=None, embedder=None, model=None, weight_of_evidence=True, memory=None):
    
    pipeline_steps = []
    
    pipeline_steps.append(make_feature_pipeline_steps(weight_of_evidence))
    
    # TODO Sampling should be done before features, but the imblearn library only supports numeric arrays. >_>
    pipeline_steps.append(('sampler', copy.deepcopy(sampler)))
    
    return Pipeline(pipeline_steps, memory=memory)


def make_feature_pipeline_steps(weight_of_evidence=True):
    feature_steps = [
        ('boolean', Pipeline([
            ('selector', TypeSelector('bool'))
        ])),

        ('numericals', Pipeline([
            ('selector', TypeSelector(np.number)),
            ('imputer', Imputer(strategy='median')),
            ('scaler', StandardScaler()),  # Careful for outliers
        ])),  
        
        ('low_dim_nominals', Pipeline([
            ('selector', ColumnSelector(columns=low_dim_categoricals)),
            ('indexer', StringIndexer())
        ]))
    ]

    return ('features', FeatureUnion(feature_steps))

        c=colors,
        s=s
    )

def plot_neighbour_similarity(X, y):
    ps = []
    rs = []
    ns = range(1,19,2)

    for n in ns:
        knn = KNN(n_neighbors=n)
        knn.fit(X, y)
        preds = knn.predict(umap_embedding)

        c = confusion_matrix(y, preds)
        print(c)
        precision = c[1,1] / c[:,1].sum()
        ps.append(precision)

        recall = c[1,1] / c[1,:].sum()
        rs.append(recall)

    # Warning: inset scores
    plt.figure(figsize=(8,5))
    plt.plot(ns, ps, label='Precision')
    plt.plot(ns, rs, label='Recall')
    plt.legend()
    plt.xlabel('Nearest Neighbors')
    plt.show()

# Cluster statistics
from scipy.stats import hypergeom
import statsmodels.stats.multitest as multitest

def cluster_p_val(y):
    """One sided fisher exact test for statistical significance of categoricals"""
    return 1 - hypergeom.cdf(
        y.sum(), # x: Nr of positives in cluster
        len(y_enc_plot), # M: Population size
        sum(y_enc_plot), # n: Total nr of positives
        len(y) # N: cluster size
    )

cluster_data = pd.DataFrame({
    'class_label': y_enc, 
    'cluster_label': cluster_labels
})

default_ratio = y_enc.mean()

cluster_stats = cluster_data \
    .groupby('cluster_label') \
    .agg({
        'class_label': {
#             'positive_ratio': lambda s: s.mean(),
#             'negative_ratio': lambda s: 1 - s.mean(),
            'relative_positive_ratio': (lambda s: s.mean() / y_enc.mean()),
            'relative_negative_ratio': (lambda s: (1 - s.mean()) / (1 - y_enc.mean())),
            'number_of_points': lambda s: len(s),
            'class_p_value': cluster_p_val
        }
    })['class_label'] \
    .sort_values(['class_p_value', 'relative_positive_ratio'], ascending=[True, False])

# Correction for multiple testing
_, cluster_stats['class_p_value'], _, _ = \
    multitest.multipletests(cluster_stats['class_p_value'], method='holm')

print("Default positive ratio:", default_ratio)
display(cluster_stats[cluster_stats['class_p_value'] <= 0.05])
